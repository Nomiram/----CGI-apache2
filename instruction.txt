#Найти <Directory /var/www/> и заменить вот этим:
###############
<Directory /var/www/>
        Options Indexes FollowSymLinks
        AllowOverride None
        Require all granted
        Options +ExecCGI
        AddHandler cgi-script .cgi

</Directory>
LoadModule cgi_module /usr/lib/apache2/modules/mod_cgi.so
################

/etc/init.d/apache2 start 
#(или reload, если уже запущен)
